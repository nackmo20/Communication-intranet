  function serializeDebugValue(value) {

    if (typeof value === "string") return value;

    try {
      return JSON.stringify(value);
    } catch {
      return String(value);
    }
  }

  function sleep(ms) {

  function normalize(text) {
    return String(text || "").replace(/\s+/g, " ").trim();
  }

  function normalizeForCompare(text) {
    return normalize(text)
      .toLocaleLowerCase("fr-FR")
      .normalize("NFD")

  }

  function escapeHtml(value) {
    return String(value ?? "")
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#039;");
  }

  function absoluteUrl(url) {
    if (!url) return "";
    try {
      return new URL(url, window.location.origin).href;
    } catch {
      return url;
    }
  }

  function describeElement(element) {
    if (!element) return "";

    const parts = [element.tagName?.toLowerCase() || "element"];
    if (element.id) parts.push(`#${element.id}`);
    if (element.name) parts.push(`[name="${element.name}"]`);
    if (element.getAttribute?.("data-drupal-selector")) parts.push(`[data-drupal-selector="${element.getAttribute("data-drupal-selector")}"]`);
    if (element.value) parts.push(`[value="${element.value}"]`);
    if (element.href) parts.push(`[href="${element.getAttribute("href")}"]`);

    const text = normalize(element.textContent || element.getAttribute?.("aria-label") || "").slice(0, 80);
    if (text) parts.push(`texte="${text}"`);

    return parts.join("");
  }

  async function waitForElement(selector, timeout = CONFIG.waits.defaultTimeout) {
    const start = Date.now();

    while (Date.now() - start < timeout) {
      const element = document.querySelector(selector);
      if (element) return element;
      await sleep(CONFIG.waits.pollInterval);
    }

    throw new Error(`Élément introuvable après ${timeout} ms : ${selector}`);
  }

      element.scrollIntoView?.();

    const previousBoxShadow = element.style.boxShadow;
    element.style.outline = "4px solid #ef4444";
    element.style.boxShadow = "0 0 0 8px rgba(239, 68, 68, .28)";

    const marker = doc.createElement("div");
    marker.textContent = `Clic script : ${label || describeElement(element)}`;
    marker.style.cssText = [
      "position:fixed",
      "left:16px",
      "top:16px",
      "z-index:2147483647",
      "padding:10px 12px",
      "border-radius:10px",

      "font:700 13px system-ui, sans-serif",
      "box-shadow:0 8px 24px rgba(0,0,0,.25)",
      "max-width:calc(100vw - 32px)"
    ].join(";");
    doc.body.appendChild(marker);

    log(`Visualisation du clic : ${label || describeElement(element)}`, element);
    await sleep(CONFIG.debug.clickPauseMs);

    marker.remove();
    element.style.outline = previousOutline;
    element.style.boxShadow = previousBoxShadow;
    win.focus?.();
  }

  function extractText(doc, selector) {
    return normalize(doc.querySelector(selector)?.textContent || "");
  }

  function extractInputValue(doc, selector) {
    const element = doc.querySelector(selector);
    if (!element) return "";
    return normalize(element.getAttribute("value") || element.value || "");
  }

  function extractTextareaValue(doc, selector) {
    const element = doc.querySelector(selector);
    if (!element) return "";
    return normalize(element.value || element.textContent || "");
  }

  function extractSelectedOptions(doc, selector) {
    const select = doc.querySelector(selector);
    if (!select) return "";

    return Array.from(select.querySelectorAll("option:checked"))
      .map((option) => normalize(option.textContent) || normalize(option.value))
      .filter(Boolean)
      .join("; ");
  }

  function extractCheckedValues(doc, selector) {
    return Array.from(doc.querySelectorAll(selector))
      .filter((input) => input.checked || input.hasAttribute("checked"))
      .map((input) => {
        const id = input.getAttribute("id");
        const label = id ? doc.querySelector(`label[for="${cssEscape(id)}"]`) : null;
        return normalize(label?.textContent) || normalize(input.getAttribute("value"));
      })
      .filter(Boolean)
      .join("; ");
  }

  function cssEscape(value) {
    if (window.CSS && typeof window.CSS.escape === "function") return window.CSS.escape(value);
    return String(value).replace(/[^a-zA-Z0-9_-]/g, "\\$&");
  }

  function valueFromFieldElement(element) {
    if (!element) return "";

    if (element.matches("textarea")) return normalize(element.value || element.textContent);

    if (element.matches("select")) {
      return Array.from(element.querySelectorAll("option:checked"))
        .map((option) => normalize(option.textContent) || normalize(option.value))
        .filter(Boolean)
        .join("; ");
    }

    if (element.matches('input[type="checkbox"], input[type="radio"]')) {
      if (!element.checked && !element.hasAttribute("checked")) return "";
      const id = element.getAttribute("id");
      const label = id ? element.ownerDocument.querySelector(`label[for="${cssEscape(id)}"]`) : null;
      return normalize(label?.textContent) || normalize(element.getAttribute("value"));
    }

    if (element.matches("input")) return normalize(element.getAttribute("value") || element.value || "");

    return normalize(element.textContent);
  }

  function extractDrupalFieldByLabel(doc, labelText) {
    const expectedLabel = normalizeForCompare(labelText);
    const labels = Array.from(doc.querySelectorAll("label, .form-item__label, .fieldset__label, legend, summary"));

    const label = labels.find((candidate) => normalizeForCompare(candidate.textContent).includes(expectedLabel));
    if (!label) return "";

    const forId = label.getAttribute("for");
    if (forId) {
      const target = doc.getElementById(forId);
      const directValue = valueFromFieldElement(target);
      if (directValue) return directValue;
    }

    const container = label.closest(".form-item, .js-form-item, .field--type-entity-reference, fieldset, details, tr, div") || label.parentElement;
    if (!container) return "";

    const checkedValues = Array.from(container.querySelectorAll('input[type="checkbox"], input[type="radio"]'))
      .filter((input) => input.checked || input.hasAttribute("checked"))
      .map((input) => valueFromFieldElement(input))
      .filter(Boolean)
      .join("; ");
    if (checkedValues) return checkedValues;

    const fields = Array.from(container.querySelectorAll("textarea, select, input:not([type='hidden']):not([type='submit']):not([type='button'])"));
    const values = fields.map((field) => valueFromFieldElement(field)).filter(Boolean);
    if (values.length) return Array.from(new Set(values)).join("; ");

    const staticValue = normalize(container.textContent).replace(normalize(label.textContent), "").trim();
    return staticValue;
  }

  function extractFirstAvailableField(doc, selectors, labels) {
    for (const selector of selectors || []) {
      const element = doc.querySelector(selector);
      const value = valueFromFieldElement(element);
      if (value) return value;
    }

    for (const label of labels || []) {
      const value = extractDrupalFieldByLabel(doc, label);
      if (value) return value;
    }

    return "";
  }

      checkbox.addEventListener("change", refreshSafetyDisplay);
      cancelButton.addEventListener("click", () => close(false));
      confirmButton.addEventListener("click", () => {
        state.showClickVisualization = Boolean(showClicksCheckbox.checked);
        state.includeDebugLogsInExcel = Boolean(includeLogsCheckbox.checked);
        if (!checkbox.checked) warn("Mode sécurité désactivé : lancement direct après confirmation utilisateur.");
        close(true);
      });
      overlay.addEventListener("click", (event) => {
        if (event.target === overlay) close(false);
      });

      refreshSafetyDisplay();
      confirmButton.focus();
    });
  }

  async function startExtractionWorkflow() {
    const button = document.getElementById("tm-drupal-extract-button");
    const progress = document.getElementById("tm-drupal-extract-progress");

      const confirmed = await confirmBeforeRun(ACTIONS_PREVIEW);

  function isForbiddenActionElement(element) {
    if (!element || !element.matches) return false;

    const value = normalizeForCompare(element.value || element.textContent || element.getAttribute("value"));
    const href = element.getAttribute?.("href") || "";

    return (
      CONFIG.ficheFormationOvp.forbiddenActionSelectors.some((selector) => element.matches(selector)) ||
      ["enregistrer", "supprimer", "cloner", "retirer"].includes(value) ||
      /\/(delete|clone)(?:[/?#]|$)/i.test(href)
    );
  }

  function isSafeParagraphEditButton(button) {
    if (!button || !button.matches || !button.matches("input")) return false;
    if (isForbiddenActionElement(button)) return false;

    const value = normalize(button.value || button.textContent).toLocaleLowerCase("fr-FR");
    const name = button.getAttribute("name") || "";
    const selector = button.getAttribute("data-drupal-selector") || "";

    return (
      value === "modifier" &&
      (
        name === "field_formation_body_0_edit" ||
        selector.includes("field-formation-body")
      )
    );
  }

  async function waitForElementInDocument(doc, selector, timeout = CONFIG.waits.defaultTimeout) {
    const start = Date.now();

    while (Date.now() - start < timeout) {
      const element = doc.querySelector(selector);
      if (element) return element;
      await sleep(CONFIG.waits.pollInterval);
    }

    throw new Error(`Élément introuvable après ${timeout} ms : ${selector}`);
  }

    return docOrWindow?.defaultView || null;

  async function extractFicheFormationOvpFromEditPage(docOrWindow) {
    log("Lecture de la fiche formation OVP...");

    const doc = getDocumentFromDocOrWindow(docOrWindow);
    const sourceWindow = getWindowFromDocOrWindow(docOrWindow);
    if (!doc) return { ficheFormation: "", ficheFormationHtml: "" };

    let editor = doc.querySelector(CONFIG.ficheFormationOvp.ckeditorSelector);

    if (!editor) {
      const editButton = findSafeParagraphEditButton(doc);

      if (editButton && sourceWindow) {
        log(`Bouton Modifier du paragraphe field_formation_body trouvé : ${describeElement(editButton)}`);
        doc.__tmDrupalAllowedParagraphEditButton = editButton;
        await clickWithDebug(editButton, "ouvrir le paragraphe Fiche formation OVP");

        try {
          editor = await waitForElementInDocument(doc, CONFIG.ficheFormationOvp.ckeditorSelector);
        } catch (err) {
          warn("CKEditor non détecté après le clic sécurisé; recherche d’un champ HTML existant.", err);
          await sleep(500);
        }
      } else if (editButton) {
        log("Bouton Modifier du paragraphe trouvé, mais clic ignoré sur document fetché non interactif.");
      }
    }

    if (!editor) {
      editor = findExistingFicheFormationContent(doc);
    }

    if (!editor) {
      warn("Fiche formation OVP introuvable dans la page d’édition lue.");
      return { ficheFormation: "", ficheFormationHtml: "" };
    }

    log("CKEditor détecté.");

    const ficheFormationHtml = normalizeHtml(editor.innerHTML || "");
    const ficheFormation = htmlElementToReadableText(editor);

    log(`Fiche formation OVP récupérée : ${ficheFormation.length} caractère(s) texte, ${ficheFormationHtml.length} caractère(s) HTML.`);
    return { ficheFormation, ficheFormationHtml };
  }

  function findExistingFicheFormationContent(doc) {
    const fieldContainer = doc.querySelector('[data-drupal-selector*="field-formation-body"], [id*="field-formation-body"], [class*="field--name-field-formation-body"]');
    if (fieldContainer) {
      const ckEditor = fieldContainer.querySelector(CONFIG.ficheFormationOvp.ckeditorSelector);
      if (ckEditor) return ckEditor;

      const textarea = fieldContainer.querySelector("textarea");
      if (textarea && normalize(textarea.value || textarea.textContent)) return textarea;
    }

    const textarea = doc.querySelector('textarea[name*="field_formation_body"], textarea[data-drupal-selector*="field-formation-body"]');
    if (textarea && normalize(textarea.value || textarea.textContent)) return textarea;

    return null;
  }

  function normalizeHtml(html) {
    return String(html || "").trim();
  }

  function htmlElementToReadableText(element) {
    if (!element) return "";

    if (element.matches?.("textarea")) {
      const rawValue = element.value || element.textContent || "";
      if (/<[a-z][\s\S]*>/i.test(rawValue)) {
        const doc = new DOMParser().parseFromString(`<div>${rawValue}</div>`, "text/html");
        return htmlElementToReadableText(doc.body.firstElementChild || doc.body);
      }
      return normalize(rawValue);
    }

    const clone = element.cloneNode(true);
    clone.querySelectorAll("br").forEach((br) => br.replaceWith("\n"));
    clone.querySelectorAll("h1, h2, h3, h4, h5, h6, p, li, div, section, article, tr").forEach((node) => {
      node.appendChild(element.ownerDocument.createTextNode("\n"));
    });

    return clone.textContent
      .split("\n")
      .map((line) => normalize(line))
      .filter(Boolean)
      .join("\n");
  }

  function uniqueReadableValues(values) {
    const seen = new Set();
    const uniqueValues = [];

    values.map(normalize).filter(Boolean).forEach((value) => {
      const key = normalizeForCompare(value);
      if (!seen.has(key)) {
        seen.add(key);
        uniqueValues.push(value);
      }
    });

    return uniqueValues;
  }

  function extractDynamicSelectionValues(doc, preferredSelector, labelText) {

  function findContainerByLabel(doc, labelText) {
    const expectedLabel = normalizeForCompare(labelText);
    const labels = Array.from(doc.querySelectorAll("label, .form-item__label, .fieldset__label, legend, summary, h2, h3, h4"));
    const label = labels.find((candidate) => normalizeForCompare(candidate.textContent).includes(expectedLabel));
    if (!label) return null;

    const forId = label.getAttribute("for");
    if (forId) {
      const target = doc.getElementById(forId);
      if (target) return target.closest(".form-item, .js-form-item, fieldset, details, .form-wrapper, div") || target.parentElement;
    }

    return label.closest(".form-item, .js-form-item, fieldset, details, .form-wrapper, tr, div") || label.parentElement;
  }

  function normalizeFieldLabelForMatch(text) {
    return normalizeForCompare(text)
      .replace(/[()]/g, " ")
      .replace(/s\b/g, "")
      .replace(/[^a-z0-9]+/g, " ")
      .trim();
  }

  function fieldLabelMatches(actualText, expectedText) {
    const actual = normalizeFieldLabelForMatch(actualText);
    const expected = normalizeFieldLabelForMatch(expectedText);
    const expectedWords = expected.split(" ").filter(Boolean);

    return actual.includes(expected) || expectedWords.every((word) => actual.includes(word));
  }

  function findFieldContainerByLabel(doc, labelText) {
    const labels = Array.from(doc.querySelectorAll("label, legend, h2, h3, summary, .form-item__label"))
      .filter((element) => fieldLabelMatches(element.textContent, labelText));

    for (const label of labels) {
      const container =
        label.closest("fieldset") ||
        label.closest(".field--type-entity-reference") ||
        label.closest(".js-form-item") ||
        label.closest(".form-item") ||
        label.closest(".form-wrapper") ||
        label.parentElement;

      if (container) return container;
    }

    return null;
  }

  function extractCheckedFancyTreeTitles(root) {
    if (!root) return "";

    const titles = [];
    const seenNodes = new WeakSet();
    const nodes = Array.from(root.querySelectorAll('.fancytree-node, li[role="treeitem"], li'));

    for (const node of nodes) {
      const treeItem = getFancyTreeOwnerItem(node);
      const canonicalNode = treeItem || node;

      if (seenNodes.has(canonicalNode)) continue;
      seenNodes.add(canonicalNode);

      const title = getFancyTreeTitleForNode(node, treeItem);
      if (!title) continue;

      const isChecked = isFancyTreeNodeChecked(node, treeItem);
      if (!isChecked) continue;

      // Règle de priorité FancyTree : quand un parent est coché, ses enfants ne
      // doivent pas être exportés en plus. On conserve donc uniquement le
      // nœud coché le plus haut dans chaque branche de l’arbre.
      if (hasCheckedFancyTreeAncestor(treeItem || node, root)) {
        log(`Fancytree : nœud ignoré car un parent prioritaire est coché : ${normalize(title.textContent)}`);
        continue;
      }

      const value = normalize(title.textContent);
      if (value) titles.push(value);
    }

    const result = uniqueReadableValues(titles).join(" | ");
    if (result) log(`Fancytree : titres cochés récupérés avec priorité parent : ${result}`);
    return result;
  }

  function getFancyTreeOwnerItem(node) {
    return node?.matches?.('li, [role="treeitem"]') ? node : node?.closest?.('li, [role="treeitem"]');
  }

  function getFancyTreeVisualNode(node, treeItem) {
    if (node?.classList?.contains("fancytree-node")) return node;
    return treeItem?.querySelector?.(":scope > .fancytree-node") || treeItem?.querySelector?.(".fancytree-node") || node;
  }

  function getFancyTreeTitleForNode(node, treeItem) {
    const visualNode = getFancyTreeVisualNode(node, treeItem);
    return visualNode?.querySelector?.(":scope > .fancytree-title") || visualNode?.querySelector?.(".fancytree-title") || node?.querySelector?.(".fancytree-title");
  }

  function isFancyTreeNodeChecked(node, treeItem, options = {}) {
    const { allowActiveFallback = true } = options;
    const visualNode = getFancyTreeVisualNode(node, treeItem);
    const checkbox = visualNode?.querySelector?.('.fancytree-checkbox, [role="checkbox"]') || node?.querySelector?.('.fancytree-checkbox, [role="checkbox"]');

    const nodeClass = visualNode?.getAttribute?.("class") || node?.getAttribute?.("class") || "";
    const checkboxClass = checkbox?.getAttribute("class") || "";
    const treeItemClass = treeItem?.getAttribute("class") || "";

    const explicitlyChecked =
      visualNode?.classList?.contains("fancytree-selected") ||
      visualNode?.classList?.contains("fancytree-checked") ||
      treeItem?.classList?.contains("fancytree-selected") ||
      treeItem?.classList?.contains("fancytree-checked") ||
      checkbox?.getAttribute("aria-checked") === "true" ||
      checkbox?.getAttribute("aria-selected") === "true" ||
      visualNode?.getAttribute?.("aria-checked") === "true" ||
      visualNode?.getAttribute?.("aria-selected") === "true" ||
      treeItem?.getAttribute("aria-checked") === "true" ||
      treeItem?.getAttribute("aria-selected") === "true" ||
      /fancytree-selected|fancytree-checked/.test(nodeClass) ||
      /fancytree-selected|fancytree-checked/.test(treeItemClass) ||
      /fancytree-selected|fancytree-checked/.test(checkboxClass);

    if (explicitlyChecked) return true;

    // Fallback conservateur : certains rendus Drupal marquent seulement le
    // nœud courant en active. On ne l’utilise pas pour tester les ancêtres afin
    // de ne pas confondre focus clavier et vraie sélection.
    return Boolean(allowActiveFallback && (
      visualNode?.classList?.contains("fancytree-active") ||
      treeItem?.classList?.contains("fancytree-active") ||
      /fancytree-active/.test(nodeClass) ||
      /fancytree-active/.test(treeItemClass)
    ));
  }

  function hasCheckedFancyTreeAncestor(node, root) {
    let ancestor = node?.parentElement?.closest?.('li[role="treeitem"], li');

    while (ancestor && ancestor !== root && root.contains(ancestor)) {
      if (isFancyTreeNodeChecked(ancestor, ancestor, { allowActiveFallback: false })) return true;
      ancestor = ancestor.parentElement?.closest?.('li[role="treeitem"], li');
    }

    return false;
  }

      };

  }

  function collectSelectionValuesFromContainer(container, labelText = "") {
    if (!container) return [];

    const rawValues = [];
    const labelCompare = normalizeForCompare(labelText);

    rawValues.push(container.getAttribute?.("title"));

    if (container.matches?.("input, textarea, select")) {
      rawValues.push(valueFromFieldElement(container));
    }

    if (container.matches?.("select")) {
      rawValues.push(...Array.from(container.querySelectorAll("option:checked")).map((option) => option.textContent || option.value));
    }

    rawValues.push(...Array.from(container.querySelectorAll("option:checked")).map((option) => option.textContent || option.value));
    rawValues.push(...Array.from(container.querySelectorAll('input:not([type="hidden"]):not([type="submit"]):not([type="button"]):not([type="checkbox"]):not([type="radio"])')).map((input) => input.value || input.getAttribute("value")));
    rawValues.push(...Array.from(container.querySelectorAll('input[type="checkbox"]:checked, input[type="radio"]:checked')).map((input) => {
      const id = input.getAttribute("id");
      const label = id ? container.ownerDocument.querySelector(`label[for="${cssEscape(id)}"]`) : null;
      return label?.textContent || input.value || input.getAttribute("value");
    }));

    // Fancytree : lire uniquement les titres réellement cochés/sélectionnés.
    rawValues.push(...extractCheckedFancyTreeTitles(container).split(" | ").filter(Boolean));

    // Widgets de sélection classiques : tokens sélectionnés uniquement, pas tous les div/span du conteneur.
    rawValues.push(...Array.from(container.querySelectorAll(".chosen-choices li.search-choice, .select2-selection__choice, .ui-autocomplete-multiselect-item, .form-autocomplete + ul li"))
      .map((element) => element.getAttribute("title") || element.textContent)
      .filter((value) => normalizeForCompare(value) !== labelCompare));

    return uniqueReadableValues(rawValues)
      .filter((value) => {
        const compared = normalizeForCompare(value);
        return compared && compared !== labelCompare && compared !== "modifier" && compared !== "retirer" && compared !== "supprimer";
      });
  }

  async function readEditPageDetails(editUrl) {
    if (!editUrl) throw new Error("Lien Modifier introuvable.");

    let fetchDetails = null;
    let fetchError = null;

    if (CONFIG.preferFetchForEditPages) {
      try {
        const response = await fetch(editUrl, { credentials: "include" });
        if (!response.ok) throw new Error(`Lecture HTTP impossible (${response.status}) pour ${editUrl}`);

        const html = await response.text();
        const doc = new DOMParser().parseFromString(html, "text/html");
        fetchDetails = await extractDetailsFromEditDocument(doc, editUrl);

        if (!needsInteractiveReadFallback(fetchDetails)) {
          return fetchDetails;
        }

        warn(`Données OVP incomplètes dans le HTML fetché pour ${editUrl}; tentative iframe en lecture seule.`);
      } catch (err) {
        fetchError = err;
        warn(`Fetch impossible ou incomplet pour ${editUrl}.`, err);
      }
    }

    if (CONFIG.allowReadOnlyIframeFallback) {
      try {
        const iframeDetails = await readEditPageByIframeFallback(editUrl);
        return mergeDetails(fetchDetails, iframeDetails);
      } catch (err) {
        warn(`Lecture iframe impossible pour ${editUrl}; conservation des données fetch disponibles.`, err);
        if (fetchDetails) return fetchDetails;
        throw err;
      }
    }

    if (CONFIG.allowReadOnlyNavigationFallback) {
      const navigationDetails = await readEditPageByNavigationFallback(editUrl);
      return mergeDetails(fetchDetails, navigationDetails);
    }

    if (fetchDetails) return fetchDetails;
    throw fetchError || new Error("Lecture par fetch désactivée et aucun secours de lecture activé.");
  }

  function needsInteractiveReadFallback(details) {
    if (!details) return true;
    return !details.ficheFormation || !details.publics || !details.departmentCodes;
  }

  function mergeDetails(primaryDetails, fallbackDetails) {
    if (!primaryDetails) return fallbackDetails;
    if (!fallbackDetails) return primaryDetails;

    return {
      ...primaryDetails,
      ...Object.fromEntries(
        Object.entries(fallbackDetails).filter(([, value]) => normalize(value))
      )
    };
  }

  async function expandSelectionTreesBeforeRead(docOrWindow) {
    const doc = getDocumentFromDocOrWindow(docOrWindow);
    const sourceWindow = getWindowFromDocOrWindow(docOrWindow);
    if (!doc || !sourceWindow) return;

    for (const labelText of ["Métiers", "Zone géographique"]) {
      await expandTreeForLabel(doc, labelText);
    }
  }

  async function expandTreeForLabel(doc, labelText) {
    log(`Déroulage de l’arbre "${labelText}" avant lecture...`);

    const container = findTreeContainerByLabel(doc, labelText) || findContainerByLabel(doc, labelText) || doc;
    const clickedControls = new WeakSet();
    let totalClicks = 0;

    await openClosedAncestorDetails(container, labelText);

    for (let pass = 0; pass < CONFIG.debug.treeExpandPasses; pass += 1) {
      const controls = findCollapsedTreeControls(container)
        .filter(isSafeTreeExpandControl)
        .filter((control) => !clickedControls.has(control));

      if (!controls.length) break;

      log(`Arbre "${labelText}" : passe ${pass + 1}, ${controls.length} nouveau(x) contrôle(s) à ouvrir.`);

      for (const control of controls) {
        if (totalClicks >= CONFIG.debug.maxTreeExpandClicks) {
          warn(`Limite de ${CONFIG.debug.maxTreeExpandClicks} clics atteinte pour l’arbre "${labelText}"; poursuite de l’export avec les nœuds déjà ouverts.`);
          log(`Déroulage de l’arbre "${labelText}" terminé après limite : ${totalClicks} clic(s).`);
          return;
        }

        clickedControls.add(control);
        await clickWithDebug(control, `dérouler ${labelText}`);
        totalClicks += 1;
        await sleep(CONFIG.debug.afterTreeExpandClickDelay);
      }
    }

    log(`Déroulage de l’arbre "${labelText}" terminé : ${totalClicks} clic(s) d’ouverture.`);
  }

  function findTreeContainerByLabel(doc, labelText) {
    const labelContainer = findContainerByLabel(doc, labelText);
    if (!labelContainer) return null;

    if (labelContainer.querySelector?.('.fancytree-expander, [role="tree"], .fancytree-container')) {
      return labelContainer;
    }

    let current = labelContainer;
    while (current && current !== doc) {
      if (current.querySelector?.('.fancytree-expander, [role="tree"], .fancytree-container')) {
        return current;
      }
      current = current.parentElement;
    }

    const labelledBy = labelContainer.querySelector?.('[aria-controls]')?.getAttribute('aria-controls');
    if (labelledBy) {
      const controlled = doc.getElementById(labelledBy);
      if (controlled) return controlled;
    }

    return labelContainer;
  }

  async function openClosedAncestorDetails(container, labelText) {
    const details = [];
    let current = container;

    const rootDocument = container.ownerDocument || document;

    while (current && current !== rootDocument) {
      if (current.matches?.("details:not([open])")) details.push(current);
      current = current.parentElement;
    }

    for (const detail of details.reverse()) {
      const summary = detail.querySelector("summary");
      if (summary && !isForbiddenActionElement(summary)) {
        await clickWithDebug(summary, `ouvrir section ${labelText}`);
        await sleep(CONFIG.debug.afterTreeExpandClickDelay);
      }
    }
  }

  function findCollapsedTreeControls(container) {
    return Array.from(container.querySelectorAll([
      '[aria-expanded="false"]',
      'button[aria-label*="Déplier"]',
      'button[aria-label*="déplier"]',
      'button[title*="Déplier"]',
      'button[title*="déplier"]',
      '.fancytree-node:not(.fancytree-expanded) > span[role="button"].fancytree-expander',
      '.fancytree-node:not(.fancytree-expanded) > .fancytree-expander',
      'span[role="button"].fancytree-expander',
      '.jstree-closed > .jstree-ocl',
      '.jstree-closed > a',
      '.tree-item--collapsed > button',
      '.tree-item--collapsed > a',
      '.tree-toggle',
      '.js-tree-toggle'
    ].join(",")));
  }

  function isSafeTreeExpandControl(control) {
    if (!control || isForbiddenActionElement(control)) return false;

    const normalizedText = normalizeForCompare(control.textContent || control.getAttribute?.("aria-label") || control.getAttribute?.("title") || "");
    const href = control.getAttribute?.("href") || "";
    const fancytreeNode = control.closest?.(".fancytree-node");

    if (fancytreeNode?.classList.contains("fancytree-expanded")) return false;
    if (fancytreeNode?.classList.contains("fancytree-loading")) return false;
    if (["enregistrer", "supprimer", "cloner", "retirer", "modifier"].some((word) => normalizedText.includes(word))) return false;
    if (/\/(delete|clone|edit)(?:[/?#]|$)/i.test(href)) return false;

    return true;
  }

  async function extractDetailsFromEditDocument(docOrWindow, editUrl) {
    const doc = getDocumentFromDocOrWindow(docOrWindow);
    const nodeInfo = getNodeInfoFromDocument(doc, editUrl);
    const ficheFormationOvp = await extractFicheFormationOvpFromEditPage(docOrWindow);

    try {
      await expandSelectionTreesBeforeRead(docOrWindow);
    } catch (err) {
      warn("Déroulage des arbres Métiers/Zone géographique impossible; poursuite de l’extraction sans bloquer la fiche OVP.", err);
    }

    const thematics = extractThematics(doc) || extractFirstAvailableField(doc, CONFIG.editFieldSelectors.thematics, CONFIG.editFieldLabels.thematics);
    const publics = extractMetiers(doc) || extractFirstAvailableField(doc, CONFIG.editFieldSelectors.publics, CONFIG.editFieldLabels.publics);
    let departmentCodes = extractZoneGeographique(doc) || extractFirstAvailableField(doc, CONFIG.editFieldSelectors.departmentCodes, CONFIG.editFieldLabels.departmentCodes);

    if (publics && departmentCodes && normalizeForCompare(publics) === normalizeForCompare(departmentCodes)) {
      warn("Métiers et zone géographique identiques détectés; la colonne Zone géographique est vidée pour éviter d’y recopier le métier.");
      departmentCodes = "";
    }

    return {
      nodeId: nodeInfo.nodeId,
      pageUrl: nodeInfo.pageUrl,
      title: extractInputValue(doc, '[name="title[0][value]"]') || extractInputValue(doc, "#edit-title-0-value") || extractText(doc, "h1"),
      chapo: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.chapo, CONFIG.editFieldLabels.chapo),
      ficheFormation: ficheFormationOvp.ficheFormation,
      ficheFormationHtml: ficheFormationOvp.ficheFormationHtml,
      thematics,
      publics,
      departmentCodes,
      preinscriptionStartDate: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.preinscriptionStartDate, CONFIG.editFieldLabels.preinscriptionStartDate),
      unpublishDate: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.unpublishDate, CONFIG.editFieldLabels.unpublishDate),
      drupalParcoursId: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.drupalParcoursId, CONFIG.editFieldLabels.drupalParcoursId),
      sourceOvpId: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.sourceOvpId, CONFIG.editFieldLabels.sourceOvpId),
      planTitle: extractFirstAvailableField(doc, CONFIG.editFieldSelectors.planTitle, CONFIG.editFieldLabels.planTitle)
    };
  }

  async function readEditPageByIframeFallback(editUrl) {
    log(`Lecture iframe contrôlée de ${editUrl}`);

    const iframe = document.createElement("iframe");
    iframe.setAttribute("aria-hidden", state.showClickVisualization ? "false" : "true");
    iframe.style.cssText = state.showClickVisualization
      ? `position:fixed;left:0;top:0;width:${CONFIG.debug.visibleIframeWidth};height:${CONFIG.debug.visibleIframeHeight};border:4px solid #ef4444;border-radius:0;background:#fff;z-index:2147483000;box-shadow:none;`
      : "position:fixed;left:0;top:0;width:1200px;height:900px;border:0;opacity:0.01;pointer-events:none;z-index:1;";

    try {
      const loaded = new Promise((resolve, reject) => {
        const timeoutId = window.setTimeout(() => reject(new Error("Chargement iframe trop long.")), CONFIG.waits.defaultTimeout);
        iframe.addEventListener("load", () => {
          window.clearTimeout(timeoutId);
          resolve();
        }, { once: true });
      });

      iframe.src = editUrl;
      document.body.appendChild(iframe);
      await loaded;
      log("Iframe de lecture chargée.");

      const doc = iframe.contentDocument;
      const frameWindow = iframe.contentWindow;
      if (!doc || !frameWindow) throw new Error("Document iframe inaccessible.");

      installReadOnlySafetyGuards(doc);
      await waitForIframeReady(doc);

      return await extractDetailsFromEditDocument(frameWindow, editUrl);
    } finally {
      iframe.remove();
    }
  }

  async function waitForIframeReady(doc) {
    const start = Date.now();

    while (Date.now() - start < CONFIG.waits.defaultTimeout) {
      if (doc.querySelector("form, main, body")) return;
      await sleep(CONFIG.waits.pollInterval);
    }

    throw new Error("La page Modifier chargée en iframe n’est pas lisible.");
  }

  function installReadOnlySafetyGuards(doc) {
    doc.addEventListener("click", (event) => {
      const target = event.target?.closest?.("input, button, a");
      if (isForbiddenActionElement(target)) {
        event.preventDefault();
        event.stopImmediatePropagation();
        warn("Action destructive bloquée en iframe.", target);
      }
    }, true);

    doc.addEventListener("submit", (event) => {
      const submitter = event.submitter || doc.activeElement || doc.__tmDrupalAllowedParagraphEditButton;
      if (submitter !== doc.__tmDrupalAllowedParagraphEditButton && !isSafeParagraphEditButton(submitter)) {
        event.preventDefault();
        event.stopImmediatePropagation();
        warn("Soumission de formulaire bloquée en iframe.", submitter);
      }
    }, true);
  }

  async function readEditPageByNavigationFallback(editUrl) {
    // Secours optionnel : ouvre un onglet/fenêtre, lit le DOM, puis ferme.
    // Aucune saisie, aucun clic sur le formulaire, aucune soumission.
    // Non utilisé par défaut pour réduire au maximum les risques.
    const editWindow = window.open(editUrl, "_blank", "width=1200,height=900");
    if (!editWindow) throw new Error("Ouverture de la page Modifier bloquée par le navigateur.");

    try {
      await waitForWindowDocument(editWindow);
      return await extractDetailsFromEditDocument(editWindow, editUrl);
    } finally {
      editWindow.close();
    }
  }

  async function waitForWindowDocument(targetWindow, timeout = CONFIG.waits.defaultTimeout) {
    const start = Date.now();

    while (Date.now() - start < timeout) {
      try {
        const doc = targetWindow.document;
        if (doc && doc.readyState === "complete" && doc.querySelector("form, main, body")) return doc;
      } catch {
        // Accès refusé tant que la page n’est pas disponible ou si la politique navigateur bloque.
      }
      await sleep(CONFIG.waits.pollInterval);
    }

    throw new Error("Impossible de lire la fenêtre Modifier en mode secours.");
  }

  function getNodeInfoFromDocument(doc, fallbackUrl) {
    const nodeLinks = [...doc.querySelectorAll("a[href], a[data-drupal-link-system-path]")];
    const paths = nodeLinks
      .flatMap((link) => [link.getAttribute("data-drupal-link-system-path"), link.getAttribute("href")])
      .filter(Boolean)
      .concat([fallbackUrl]);

    const nodeId = paths.map((path) => path.match(/(?:^|\/)node\/(\d+)(?:\/|$|[?#])/)?.[1]).find(Boolean) || "";
    const viewLink =
      nodeLinks.find((link) => link.getAttribute("data-drupal-link-system-path") === `node/${nodeId}`) ||
      nodeLinks.find((link) => normalizeForCompare(link.textContent) === "voir" && link.getAttribute("href"));

    return {
      nodeId,
      pageUrl: absoluteUrl(viewLink?.getAttribute("href") || (nodeId ? `/node/${nodeId}` : ""))
    };
  }

  function getCurrentNodeInfo() {
    const nodeLinks = [...document.querySelectorAll("a[href], a[data-drupal-link-system-path]")];
    const paths = nodeLinks
      .flatMap((link) => [link.getAttribute("data-drupal-link-system-path"), link.getAttribute("href")])
      .filter(Boolean);

    const nodeId = paths.map((path) => path.match(/(?:^|\/)node\/(\d+)(?:\/|$|[?#])/)?.[1]).find(Boolean) || "";
    const viewLink =
      nodeLinks.find((link) => link.getAttribute("data-drupal-link-system-path") === `node/${nodeId}`) ||
      nodeLinks.find((link) => normalize(link.textContent).toLocaleLowerCase("fr-FR") === "voir" && link.getAttribute("href"));

    const pageUrl = viewLink?.href || (nodeId && /\/node\/\d+/.test(window.location.pathname) ? window.location.href : "");
    return { nodeId, pageUrl };
  }

  function exportReadRowsToExcel(rows) {
    if (!rows.length) {
      alert("Aucune ligne lue à exporter pour le moment.");
      return;
    }

    const html = `
      <!doctype html>
      <html>
        <head>
          <meta charset="utf-8">
          <style>td { white-space: pre-wrap; }</style>
        </head>
        <body>
          <table>
            <thead>
              <tr>${EXPORT_HEADERS.map((header) => `<th>${escapeHtml(header)}</th>`).join("")}</tr>
            </thead>
            <tbody>
              ${rows.map((row) => `
                <tr>
                  ${[
                    row.status || "Lu",
                    row.nodeId || "",
                    row.pageUrl || "",
                    row.title || "",
                    row.chapo || "",
                    row.ficheFormation || "",
                    row.thematics || "",
                    row.publics || "",
                    row.departmentCodes || "",
                    row.preinscriptionStartDate || "",
                    row.unpublishDate || "",
                    row.drupalParcoursId || "",
                    row.sourceOvpId || "",
                    row.planTitle || "",
                    row.exportedAt || ""
                  ].map((value) => `<td>${escapeHtml(value)}</td>`).join("")}
                </tr>
              `).join("")}
            </tbody>
          </table>
          ${state.includeDebugLogsInExcel ? buildDebugLogsHtmlTable() : ""}
        </body>
      </html>
    `;

    const blob = new Blob([html], { type: "application/vnd.ms-excel;charset=utf-8" });
    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");

    link.href = url;
    link.download = `${CONFIG.exportFilePrefix}-${todayIsoDate()}.xls`;

    document.body.appendChild(link);
    link.click();
    link.remove();
    URL.revokeObjectURL(url);

    log(`${rows.length} ligne(s) exportée(s) vers Excel.`);
  }
